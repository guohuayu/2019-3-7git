git init  					初始化本地仓库

git add <文件名> || .|| *文件类型		添加文件提交列表

git status					查询状态（*****）
	
git em --cached <文件名> 			删除新提交的文件

git reset  <文件名> 		

git commit -m  "备注信息" ||  :wq		提交 本地仓库

在初始化文件夹创建 touch .gitignore 文件在里面写<文件名> || /<文件夹名>  	

git checkout master || <分支名>			 	进入主分支
git branch   <分支名>				 	创建分支

git merge <分支名>				分支导入主分支（在主分支下导入）
